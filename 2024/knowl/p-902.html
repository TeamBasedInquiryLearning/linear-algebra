Let \(\vec u=a+b\mathbf{i}\text{,}\) \(\vec v=c+d\mathbf{i}\text{,}\) and \(\vec w=e+f\mathbf{i}\text{.}\) Then
\begin{align*} \vec u\oplus(\vec v \oplus \vec w) &= (a+b\mathbf{i})\oplus((c+d\mathbf{i})\oplus(e+f\mathbf{i}))\\ &= (a+b\mathbf{i})\oplus((c+e)+(d+f)\mathbf{i})\\ &=(a+c+e)+(b+d+f)\mathbf{i} \end{align*}
\begin{align*} (\vec u\oplus\vec v) \oplus \vec w &= ((a+b\mathbf{i})\oplus(c+d\mathbf{i}))\oplus(e+f\mathbf{i})\\ &=((a+c)+(b+d)\mathbf{i})\oplus(e+f\mathbf{i})\\ &=(a+c+e)+(b+d+f)\mathbf{i} \end{align*}
