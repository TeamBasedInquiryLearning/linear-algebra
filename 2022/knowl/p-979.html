\begin{equation*} \det([\cdots\hspace{0.5em}\vec{v}\hspace{0.5em} \cdots\hspace{1em}\vec{w}\hspace{0.5em} \cdots])= -\det([\cdots\hspace{0.5em}\vec{w}\hspace{0.5em} \cdots\hspace{1em}\vec{v}\hspace{0.5em} \cdots]) \end{equation*}
